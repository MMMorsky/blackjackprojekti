3   import blakjakprojekti.logiikka.Kortti;
4   import blakjakprojekti.logiikka.Pelipoyta;
5   import java.awt.image.BufferedImage;
6   import java.io.File;
7   import java.io.IOException;
8   import java.io.InputStream;
9   import java.util.ArrayList;
10  import java.util.List;
11  import java.util.logging.Level;
12  import java.util.logging.Logger;
13  import javax.imageio.ImageIO;
14  import javax.swing.ImageIcon;
15  import javax.swing.JLabel;
16  
17  public class KortinValitsin {
18  
19      private Pelipoyta pelipoyta;
20  
21      /**
22       * Tämä luokka valitsee käyttöliittymälle, oikean kuvan vastaamaan korttia.
23       * @param pelipoyta Ohjelman toiminnallisuus
24       */
25      public KortinValitsin(Pelipoyta pelipoyta) {
26          this.pelipoyta = pelipoyta;
27      }
28  
29      
30      /**
31       * Palauttaa pelaajan kortit kuvina.
32       * @return Palauttaa kortit listana JLabeleita
33       */
34      public List<JLabel> palautaPelaajanKortit() {
35          List<JLabel> pelaajankuvat = new ArrayList<>();
36  
37          BufferedImage kuva = null;
38          for (Kortti kortti : pelipoyta.getPelaaja().getKortit()) {
39              InputStream is = getClass().getClassLoader().getResourceAsStream("kortit/" + kortti.palautaTiedostopaate());
40              try {
41                  kuva = ImageIO.read(is);
42              } catch (IOException ex) {
43                  Logger.getLogger(KortinValitsin.class.getName()).log(Level.SEVERE, null, ex);
44              }
45  
46              JLabel picLabel = new JLabel(new ImageIcon(kuva));
47              pelaajankuvat.add(picLabel);
48          }
49  
50          return pelaajankuvat;
51      }
52  
53      /**
54       * Palauttaa jakajankortit kuvina.
55       * @return Palautuu JLabel listana
56       */
57      public List<JLabel> palautaJakajanKortit() {
58          List<JLabel> jakajankuvat = new ArrayList<>();
59  
60          BufferedImage kuva = null;
61          for (Kortti kortti : pelipoyta.getJakaja().getKortit()) {
62              InputStream is = getClass().getClassLoader().getResourceAsStream("kortit/" + kortti.palautaTiedostopaate());
63              try {
64                  kuva = ImageIO.read(is);
65              } catch (IOException ex) {
66                  Logger.getLogger(KortinValitsin.class.getName()).log(Level.SEVERE, null, ex);
67              }
68  
69              JLabel picLabel = new JLabel(new ImageIcon(kuva));
70              jakajankuvat.add(picLabel);
71          }
72  
73          return jakajankuvat;
74      }
75  }
