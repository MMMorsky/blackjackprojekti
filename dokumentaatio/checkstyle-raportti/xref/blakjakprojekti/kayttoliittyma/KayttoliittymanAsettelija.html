1   package blakjakprojekti.kayttoliittyma;
2   
3   import blakjakprojekti.logiikka.Pelipoyta;
4   import java.awt.Container;
5   import java.awt.Font;
6   import java.awt.GridLayout;
7   import javax.swing.JButton;
8   import javax.swing.JLabel;
9   import javax.swing.JPanel;
10  
11  public class KayttoliittymanAsettelija {
12  
13      private Container container;
14      private Pelipoyta pelipoyta;
15  
16      
17      /**
18       * Luo käyttöliittymän asettelijan.
19       * @param pelipoyta 
20       */
21      public KayttoliittymanAsettelija(Pelipoyta pelipoyta) {
22          this.pelipoyta = pelipoyta;
23      }
24      
25      /**
26       * Asettelee käyttöliittymän komponentit.
27       * @return Palauttaa käyttöliittymän komponentit.
28       */
29      public JPanel asettele() {
30  
31          JPanel p = new JPanel(new GridLayout(0, 10));
32          for (int i = 0; i < 100; i++) {
33              if (i == 0) {
34                  p.add(luoAloitaPainike(container));
35              } else if (i == 90) {
36                  p.add(luoNostaPainike(container));
37              } else if (i == 92) {
38                  p.add(luoJaaPainike(container));
39              } else {
40                  p.add(new JLabel());
41              }
42          }
43          return p;
44      }
45  
46      private JButton luoAloitaPainike(Container container) {
47          JButton painike = new JButton("Aloita");
48          painike.setFont(new Font("Arial", Font.PLAIN, 15));
49          painike.addActionListener(new AloitaKuuntelija(pelipoyta, container));
50          return painike;
51      }
52  
53      private JButton luoNostaPainike(Container container) {
54          JButton painike = new JButton("Nosta");
55          painike.setFont(new Font("Arial", Font.PLAIN, 15));
56          painike.addActionListener(new NostaKuuntelija(pelipoyta, container));
57          return painike;
58      }
59  
60      private JButton luoJaaPainike(Container container) {
61          JButton painike = new JButton("Jää");
62          painike.setFont(new Font("Arial", Font.PLAIN, 15));
63          painike.addActionListener(new JaaKuuntelija(pelipoyta, container));
64          return painike;
65      }
66  
67  }
